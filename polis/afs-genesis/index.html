AFS is a self-improving agentic CLI bootstrapped from a curl call to OpenAI Chat Completions. It uses a tool loop to execute shell commands, edits its own code, and later runs in Telegram long-poll mode.

MODEL = "gpt-5.2"
API_URL = "https://api.openai.com/v1/chat/completions"

SYSTEM_PROMPT = "You are AFS, bootstrapped from curl. Help user build systems."

def utc_ts():
    return datetime.utcnow().isoformat(timespec="seconds") + "Z"


def log_line(s: str):
    try:
        with open(LOG_PATH, "a", encoding="utf-8") as f:
            f.write(f"[{utc_ts()}] {s}\n")
    except Exception:
        # Avoid crashing on logging issues
        pass


def log_json(label: str, obj):
    try:
        payload = json.dumps(obj, ensure_ascii=False, sort_keys=True)
    except Exception:
        payload = repr(obj)
    log_line(f"{label}: {payload}")


def parse_env_file(path: str):
    env = {}
    if not os.path.exists(path):
        return env
    with open(path, "r", encoding="utf-8") as f:
        for raw in f:
            line = raw.strip()
            if not line or line.startswith("#"):
                continue
            if "=" not in line:
                continue
            k, v = line.split("=", 1)
            k = k.strip()
            v = v.strip()
            if (v.startswith('"') and v.endswith('"')) or (v.startswith("'") and v.endswith("'")):
                v = v[1:-1]
            env[k] = v
    return env


def require_api_key():
    env = parse_env_file(ENV_PATH)
    key = env.get("OPENAI_API_KEY") or os.environ.get("OPENAI_API_KEY")
    if not key:
        msg = f"Missing OPENAI_API_KEY. Put it in {ENV_PATH} as OPENAI_API_KEY=... or set env var."
        log_line("ERROR: " + msg)
        print(msg, file=sys.stderr)
        sys.exit(1)
    return key


def truncate(s: str, limit: int):
    if s is None:
        return ""
    if len(s) <= limit:
        return s
    return s[:limit] + f"\n\n[truncated to {limit} chars]"


def run_bash(command: str):
    t0 = time.time()
    log_json("tool.bash.input", {"command": command, "timeout": BASH_TIMEOUT_SECONDS})
    try:
        proc = subprocess.run(
            command,
            shell=True,
            text=True,
            capture_output=True,
            timeout=BASH_TIMEOUT_SECONDS,
            executable="/bin/bash" if os.path.exists("/bin/bash") else None,
        )
        elapsed = time.time() - t0
        out = proc.stdout or ""
        err = proc.stderr or ""
        combined = ""
        if out:
            combined += out
        if err:
            if combined and not combined.endswith("\n"):
                combined += "\n"
            combined += err
        combined = combined if combined else ""
        combined = truncate(combined, BASH_TRUNCATE_CHARS)
        result = {
            "ok": proc.returncode == 0,
            "returncode": proc.returncode,
            "elapsed_seconds": round(elapsed, 3),
            "output": combined,
        }
        log_json("tool.bash.output", result)
        return result
    except subprocess.TimeoutExpired as e:
        elapsed = time.time() - t0
        partial = ""
        if getattr(e, "stdout", None):
            partial += e.stdout
        if getattr(e, "stderr", None):
            if partial and not partial.endswith("\n"):
                partial += "\n"
            partial += e.stderr
        partial = truncate(partial, BASH_TRUNCATE_CHARS)
        result = {
            "ok": False,
            "returncode": None,
            "elapsed_seconds": round(elapsed, 3),
            "error": "timeout",
            "output": partial,
        }
        log_json("tool.bash.output", result)
        return result
    except Exception as e:
        elapsed = time.time() - t0
        result = {
            "ok": False,
            "returncode": None,
            "elapsed_seconds": round(elapsed, 3),
            "error": f"{type(e).__name__}: {e}",
            "output": "",
        }
        log_json("tool.bash.output", result)
        return result


def http_post_json(url: str, headers: dict, body_obj: dict, timeout: int = 120):
    body_bytes = json.dumps(body_obj).encode("utf-8")
    req = urllib.request.Request(url, data=body_bytes, method="POST")
    for k, v in headers.items():
        req.add_header(k, v)
    req.add_header("Content-Type", "application/json")
    req.add_header("Accept", "application/json")
    log_json("http.request", {"url": url, "headers": {k: ("***" if k.lower() == "authorization" else v) for k, v in headers.items()}, "body": body_obj})
    try:
        with urllib.request.urlopen(req, timeout=timeout) as resp:
            resp_bytes = resp.read()
            resp_text = resp_bytes.decode("utf-8", errors="replace")
            try:
                resp_obj = json.loads(resp_text)
            except Exception:
                resp_obj = {"_raw": resp_text}
            log_json("http.response", {"status": getattr(resp, "status", None), "body": resp_obj})
            return resp_obj, getattr(resp, "status", None)
    except urllib.error.HTTPError as e:
        resp_text = ""
        try:
            resp_text = e.read().decode("utf-8", errors="replace")
        except Exception:
            resp_text = ""
        try:
            resp_obj = json.loads(resp_text) if resp_text else {"error": {"message": "HTTPError with empty body"}}
        except Exception:
            resp_obj = {"_raw": resp_text}
        log_json("http.response", {"status": e.code, "body": resp_obj})
        return resp_obj, e.code
    except Exception as e:
        log_json("http.error", {"error": f"{type(e).__name__}: {e}"})
        return {"error": {"message": f"{type(e).__name__}: {e}"}}, None


def extract_assistant_message(resp_obj: dict):
    # For Chat Completions API: choices[0].message = {role, content, tool_calls?}
    try:
        choices = resp_obj.get("choices") or []
        if not choices:
            return None
        msg = choices[0].get("message")
        return msg
    except Exception:
        return None

def call_openai(api_key: str, messages: list):
    headers = {"Authorization": f"Bearer {api_key}"}
    body = {
        "model": MODEL,
        "messages": messages,
        "tools": build_tools_spec(),
        "tool_choice": "auto",
    }
    resp_obj, status = http_post_json(API_URL, headers, body)
    if status and status >= 400:
        return None, resp_obj
    msg = extract_assistant_message(resp_obj)
    if msg is None:
        return None, resp_obj
    return msg, resp_obj


def normalize_tool_calls(msg: dict):
    # Expect msg["tool_calls"] like:
    # [{"id": "...","type":"function","function":{"name":"bash","arguments":"{\"command\":\"ls\"}"}}]
    tool_calls = msg.get("tool_calls")
    if not tool_calls:
        return []
    if isinstance(tool_calls, list):
        return tool_calls
    return []


def tool_result_message(tool_call_id: str, tool_name: str, result_obj: dict):
    # Chat Completions expects role="tool", tool_call_id, content as string
    return {
        "role": "tool",
        "tool_call_id": tool_call_id,
        "content": json.dumps({"tool": tool_name, "result": result_obj}, ensure_ascii=False),
    }


def run_agent_turn(api_key: str, messages: list):
    # Full agent loop: keep calling until assistant returns without tool_calls
    while True:
        msg, raw = call_openai(api_key, messages)
        if msg is None:
            err = raw.get("error") if isinstance(raw, dict) else None
            text = ""
            if isinstance(err, dict) and err.get("message"):
                text = f"API error: {err.get('message')}"
            else:
                text = "API error: (unknown)"
            log_json("agent.error", {"raw": raw})
            messages.append({"role": "assistant", "content": text})
            return text

        log_json("agent.assistant_message", msg)
        # Append assistant message to transcript (including tool_calls metadata)
        messages.append(msg)

        tool_calls = normalize_tool_calls(msg)
        if not tool_calls:
            content = msg.get("content") or ""
            return content

        for tc in tool_calls:
            tc_id = tc.get("id") or ""
            tc_type = tc.get("type") or ""
            fn = (tc.get("function") or {}) if isinstance(tc, dict) else {}
            name = fn.get("name") or ""
            args_raw = fn.get("arguments") or ""
            log_json("agent.tool_call", {"id": tc_id, "type": tc_type, "name": name, "arguments": args_raw})

            if name != "bash":
                result = {"ok": False, "error": f"unknown tool: {name}"}
                messages.append(tool_result_message(tc_id, name, result))
                continue

            try:
                args_obj = json.loads(args_raw) if isinstance(args_raw, str) else (args_raw or {})
            except Exception as e:
                args_obj = {}
                result = {"ok": False, "error": f"invalid tool arguments JSON: {type(e).__name__}: {e}", "raw": args_raw}
                messages.append(tool_result_message(tc_id, name, result))
                continue

            command = args_obj.get("command")
            if not isinstance(command, str) or not command.strip():
                result = {"ok": False, "error": "missing required string argument: command", "args": args_obj}
                messages.append(tool_result_message(tc_id, name, result))
                continue

            result = run_bash(command)
            messages.append(tool_result_message(tc_id, name, result))

        # Continue loop: send tool results back to model


def repl():
    api_key = require_api_key()
    log_line("AFS starting")
    messages = [{"role": "system", "content": SYSTEM_PROMPT}]

    print("AFS CLI. Type /exit to quit. Type /reset to clear context.", flush=True)
    while True:
        try:
            user = input("> ")
        except EOFError:
            print()
            break
        except KeyboardInterrupt:
            print()
            break

        if user is None:
            continue
        user = user.rstrip("\n")
        if not user.strip():
            continue

        if user.strip() in ("/exit", "/quit"):
            break
        if user.strip() == "/reset":
            messages = [{"role": "system", "content": SYSTEM_PROMPT}]
            log_line("REPL reset")
            print("(context reset)", flush=True)
            continue

        user_msg = {"role": "user", "content": user}
        log_json("repl.user", user_msg)
        messages.append(user_msg)

        try:
            out = run_agent_turn(api_key, messages)
        except KeyboardInterrupt:
            log_line("Interrupted during agent turn")
            print("(interrupted)", flush=True)
            continue
        except Exception as e:
            log_json("agent.exception", {"error": f"{type(e).__name__}: {e}"})
            print(f"Error: {type(e).__name__}: {e}", file=sys.stderr, flush=True)
            continue

        out = out or ""
        log_json("repl.assistant_text", {"content": out})
        sys.stdout.write(out + ("\n" if not out.endswith("\n") else ""))
        sys.stdout.flush()

    log_line("AFS exiting")


if __name__ == "__main__":
    repl()

import os, subprocess, tempfile, time, uuid

def _head_lines(path, n_lines=20, max_bytes=8192):
    out = []
    size = 0
    try:
        with open(path, "rb") as f:
            for _ in range(n_lines):
                line = f.readline()
                if not line:
                    break
                if size + len(line) > max_bytes:
                    line = line[: max(0, max_bytes - size)]
                    out.append(line)
                    size = max_bytes
                    break
                out.append(line)
                size += len(line)
                if size >= max_bytes:
                    break
        return b"".join(out).decode("utf-8", errors="replace")
    except FileNotFoundError:
        return ""

def bash_tool(command, cwd=None, env=None, timeout_sec=600, preview_lines=20, max_preview_bytes=8192):
    env2 = os.environ.copy()
    if env:
        env2.update({k: str(v) for k, v in env.items()})

    uid = str(uuid.uuid4())[:8]
    stdout_path = os.path.join(tempfile.gettempdir(), f"afs-stdout-{uid}.log")
    stderr_path = os.path.join(tempfile.gettempdir(), f"afs-stderr-{uid}.log")

    start = time.time()
    with open(stdout_path, "wb") as so, open(stderr_path, "wb") as se:
        try:
            p = subprocess.run(
                ["bash", "-lc", command],
                cwd=cwd or None,
                env=env2,
                stdout=so,
                stderr=se,
                timeout=timeout_sec,
                text=False
            )
            exit_code = p.returncode
            timed_out = False
        except subprocess.TimeoutExpired:
            exit_code = 124
            timed_out = True

    return {
        "exit_code": exit_code,
        "timed_out": timed_out,
        "duration_sec": round(time.time() - start, 3),
        "stdout_path": stdout_path,
        "stderr_path": stderr_path,
        "stdout_preview": _head_lines(stdout_path, preview_lines, max_preview_bytes),
        "stderr_preview": _head_lines(stderr_path, preview_lines, max_preview_bytes),
    }

            if cur > end_line:
                break
            chunks.append(line)
            if sum(len(c) for c in chunks) > max_bytes:
                truncated = True
                break
        data = b"".join(chunks)[:max_bytes]

    sha = hashlib.sha256(data).hexdigest()
    return {
        "path": path,
        "start_line": start_line,
        "end_line": end_line,
        "truncated": truncated,
        "sha256": sha,
        "content": data.decode("utf-8", errors="replace"),
    }

2.3 write_file implementation (atomic)

import os, tempfile, hashlib

def write_file_tool(path, content, create_dirs=True, mode=None, atomic=True):
    b = content.encode("utf-8")
    if create_dirs:
        os.makedirs(os.path.dirname(path) or ".", exist_ok=True)

    if atomic:
        d = os.path.dirname(path) or "."
        fd, tmp = tempfile.mkstemp(prefix=".afs-tmp-", dir=d)
        try:
            with os.fdopen(fd, "wb") as f:
                f.write(b)
            if mode:
                os.chmod(tmp, int(mode, 8))
            os.replace(tmp, path)
        finally:
            try:
                if os.path.exists(tmp):
                    os.remove(tmp)
            except OSError:
                pass
    else:
        with open(path, "wb") as f:
            f.write(b)
        if mode:
            os.chmod(path, int(mode, 8))

    return {
        "path": path,
        "bytes_written": len(b),
        "sha256": hashlib.sha256(b).hexdigest()
    }

Python implementation

import os, subprocess, tempfile, uuid, time

def _head_lines(path, n_lines=20, max_bytes=8192):
    out = []
    size = 0
    try:
        with open(path, "rb") as f:
            for _ in range(n_lines):
                line = f.readline()
                if not line:
                    break
                if size + len(line) > max_bytes:
                    line = line[: max(0, max_bytes - size)]
                    out.append(line)
                    size = max_bytes
                    break
                out.append(line)
                size += len(line)
                if size >= max_bytes:
                    break
        return b"".join(out).decode("utf-8", errors="replace")
    except FileNotFoundError:
        return ""

def patch_tool(
    diff,
    cwd=None,
    strip=0,
    fail_on_mismatch=True,
    timeout_sec=60,
    preview_lines=20,
    max_preview_bytes=8192
):
    uid = str(uuid.uuid4())[:8]
    tmpdir = tempfile.gettempdir()

    diff_path   = os.path.join(tmpdir, f"afs-diff-{uid}.patch")
    stdout_path = os.path.join(tmpdir, f"afs-patch-stdout-{uid}.log")
    stderr_path = os.path.join(tmpdir, f"afs-patch-stderr-{uid}.log")
    reject_path = os.path.join(tmpdir, f"afs-patch-rejects-{uid}.rej")

    with open(diff_path, "wb") as f:
        f.write(diff.encode("utf-8"))

    # Notes:
    # --batch: never ask questions
    # --forward: ignore patches already applied (exit code can still be nonzero depending on version)
    # --reject-file: put rejects somewhere we control
    # --posix: optional; can change behavior. I usually avoid unless needed.
    cmd = [
        "patch",
        f"-p{int(strip)}",
        "--batch",
        "--reject-file", reject_path,
        "--input", diff_path,
    ]

    start = time.time()
    with open(stdout_path, "wb") as so, open(stderr_path, "wb") as se:
        try:
            p = subprocess.run(
                cmd,
                cwd=cwd or None,
                stdout=so,
                stderr=se,
                timeout=timeout_sec,
                text=False
            )
            exit_code = p.returncode
            timed_out = False
        except subprocess.TimeoutExpired:
            exit_code = 124
            timed_out = True

    stdout_preview = _head_lines(stdout_path, preview_lines, max_preview_bytes)
    stderr_preview = _head_lines(stderr_path, preview_lines, max_preview_bytes)

    rejects_exist = os.path.exists(reject_path) and os.path.getsize(reject_path) > 0

    # Interpretation:
    # patch exit codes vary a bit; commonly:
    # 0 = all hunks applied
    # 1 = some hunks failed (rejects)
    # 2 = trouble (bad diff, missing files, etc.)
    applied_ok = (exit_code == 0) or (exit_code == 1 and not fail_on_mismatch)

    return {
        "exit_code": exit_code,
        "timed_out": timed_out,
        "duration_sec": round(time.time() - start, 3),

        "applied": bool(applied_ok and not timed_out),
        "fail_on_mismatch": bool(fail_on_mismatch),

        "cwd": cwd,
        "strip": int(strip),

        "diff_path": diff_path,
        "stdout_path": stdout_path,
        "stderr_path": stderr_path,
        "rejects_path": reject_path if rejects_exist else None,

        "stdout_preview": stdout_preview,
        "stderr_preview": stderr_preview
    }

Suggested host-side policy

  • If exit_code != 0 and fail_on_mismatch=True, surface applied=false and keep rejects_path (if any).
  • The model can call read_file(rejects_path, …) to inspect rejects without dumping everything.
